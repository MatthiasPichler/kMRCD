function result = evaluation(data, labels,alpha)
    arguments
        data (:,:) double
        labels (:,1) categorical
        alpha (1,1) double {mustBeInRange(alpha,0.5,1)}
    end

    assert(isequal(size(labels), [size(data, 1), 1]))

    x = rZscores(data);
    kModel = AutoRbfKernel(x);
    poc = kMRCD(kModel);
    solution = poc.runAlgorithm(x, alpha);

    result = struct();
    result.hSubset = data(solution.hsubsetIndices, :);
    result.outliers = data(solution.flaggedOutlierIndices, :);

    fig = figure();
    mahalchart(labels, solution.rd, solution.cutoff)
    clf(fig);

    grouphat = categorical(repmat("inlier", size(labels)), {'inlier' 'outlier'});
    grouphat(solution.flaggedOutlierIndices) = "outlier";

    stats = struct();
    stats.kMRCD = accuracy(confusionmat(labels,grouphat));

    scores = struct();
    % TODO add scores for kMRCD

    % Local Outlier Factor
    [~,tf, sc] = lof(data, ContaminationFraction=(1-alpha));
    grouphat = categorical(repmat("inlier", size(labels)), {'inlier' 'outlier'});
    grouphat(tf) = "outlier";
    stats.lof = accuracy(confusionmat(labels,grouphat));
    scores.lof = sc;

    % Isolation Forest
    [~,tf, sc] = iforest(data, ContaminationFraction=(1-alpha));
    grouphat = categorical(repmat("inlier", size(labels)), {'inlier' 'outlier'});
    grouphat(tf) = "outlier";
    stats.iforest = accuracy(confusionmat(labels,grouphat));
    scores.iforest = sc;

    % Random Robust Cut Forest
    [~,tf,sc] = rrcforest(data, ContaminationFraction=(1-alpha));
    grouphat = categorical(repmat("inlier", size(labels)), {'inlier' 'outlier'});
    grouphat(tf) = "outlier";
    stats.rrcforest = accuracy(confusionmat(labels,grouphat));
    scores.rrcforest = sc;

    % One-Class Support Vector Machine
    [~,tf,sc] = ocsvm(data, ContaminationFraction=(1-alpha));
    grouphat = categorical(repmat("inlier", size(labels)), {'inlier' 'outlier'});
    grouphat(tf) = "outlier";
    stats.ocsvm = accuracy(confusionmat(labels,grouphat));
    scores.ocsvm = sc;

    clear tf;
    clear sc;

    result.stats = struct2table( ...
        vertcat(stats.kMRCD, stats.lof, ...
                stats.iforest, stats.rrcforest, ...
                stats.ocsvm), 'RowNames', fieldnames(stats));

    fig = figure();
    hold on;
    xlabel('Precision');
    ylabel('Recall');
    title('Precision-Recall Curve');

    fn = fieldnames(scores);
    colors = jet(numel(fn));
    for k=1:numel(fn)
        s = scores.(fn{k});
        [X,Y,~,AUC] = perfcurve(labels,s,'outlier','XCrit', 'prec', 'YCrit', 'reca');
        plot(X,Y, 'DisplayName', sprintf('%s (AUC=%0.4f)', fn{k}, AUC), 'Color', colors(k,:));
    end
    legend;
    hold off;
end
